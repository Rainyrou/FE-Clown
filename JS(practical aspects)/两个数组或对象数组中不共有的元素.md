###### 简单数组

```JavaScript
const find = (arr1, arr2) => {
  const set1 = new Set(arr1),
    set2 = new Set(arr2);
  const newArr1 = arr1.filter((item) => !set2.has(item)),
    newArr2 = arr2.filter((item) => !set1.has(item));
  return newArr1.concat(newArr2);
};

const arr1 = [1, 2, 3, 4],
  arr2 = [3, 4, 5, 6];
console.log(find(arr1, arr2)); 
```

###### 对象数组

```JavaScript
const find = (arr1, arr2, key) => {
  const set1 = new Set(arr1.map((item) => item[key])),
    set2 = new Set(arr2.map((item) => item[key]));
  const newArr1 = arr1.filter((item) => !set2.has(item[key])),
    newArr2 = arr2.filter((item) => !set1.has(item[key]));
  return newArr1.concat(newArr2);
};

const arr1 = [
  { id: 1, name: "Boss" },
  { id: 2, name: "Rainy" },
];
const arr2 = [
  { id: 2, name: "Rainy" },
  { id: 3, name: "Clown" },
];

console.log(find(arr1, arr2, "id"));
```
