定时器：

```js
class Timer {
  constructor(callback, interval) {
    this.timerId = null;
    this.callback = callback;
    this.interval = interval;
  }
  start() {
    if (this.timerId) return;
    this.timerId = setInterval(this.callback, this.interval);
  }
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
  }
}
```

秒表：

```js
class Watch {
  constructor(callback, precision = 100) {
    this.intervalId = null;
    this.elapsedTime = 0;
    this.callback = callback;
    this.precision = precision;
  }
  start() {
    if (this.intervalId) return;
    this.intervalId = setInterval(() => {
      this.elapsedTime += this.precision;
      this.callback(this.elapsedTime);
    }, this.precision);
  }
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
  }
  reset() {
    this.stop();
    this.elapsedTime = 0;
    this.callback(0);
  }
}
```

倒计时：

```js
class CountdownTimer {
  constructor(duration, callback, precision = 1000) {
    this.intervalId = null;
    this.initialDuration = duration;
    this.remainingTime = duration;
    this.callback = callback;
    this.precision = precision;
  }
  start() {
    if (this.intervalId) return;
    this.intervalId = setInterval(() => {
      this.remainingTime -= this.precision;
      if (this.remainingTime <= 0) {
        this.stop();
        this.remainingTime = 0;
      }
      this.callback(this.remainingTime);
    }, this.precision);
  }
  stop() {
    if (this.intervalId) {
      clearInterval(this.intervalId);
      this.intervalId = null;
    }
  }
  reset() {
    this.stop();
    this.remainingTime = this.initialDuration;
    this.callback(this.remainingTime);
  }
}
```
