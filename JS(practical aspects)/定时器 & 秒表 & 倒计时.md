定时器：

```js
class Timer {
  constructor(callback, interval) {
    this.timerId = null;
    this.callback = callback;
    this.interval = interval;
  }
  start() {
    if (this.timerId) return;
    this.timerId = setInterval(this.callback, this.interval);
  }
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
  }
}
```

秒表：

通过 `callback` 接收 `this.passedTime` 并执行自定义逻辑，实时通知外部当前秒表时间

```js
class Watch {
  constructor(callback, precision = 100) {
    this.timerId = null;
    this.passedTime = 0;
    this.callback = callback;
    this.precision = precision;
  }
  start() {
    if (this.timerId) return;
    this.timerId = setInterval(() => {
      this.passedTime += this.precision;
      this.callback(this.passedTime);
    }, this.precision);
  }
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
  }
  reset() {
    this.stop();
    this.passedTime = 0;
    this.callback(0);
  }
}
```

倒计时：

```js
class CountdownTimer {
  constructor(duration, callback, precision = 1000) {
    this.timerId = null;
    this.initialDuration = duration;
    this.remainedTime = duration;
    this.callback = callback;
    this.precision = precision;
  }
  start() {
    if (this.timerId) return;
    this.timerId = setInterval(() => {
      this.remainedTime -= this.precision;
      if (this.remainedTime <= 0) {
        this.stop();
        this.remainedTime = 0;
      }
      this.callback(this.remainedTime);
    }, this.precision);
  }
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
  }
  reset() {
    this.stop();
    this.remainedTime = this.initialDuration;
    this.callback(this.remainedTime);
  }
}
```
