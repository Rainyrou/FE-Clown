```js
class Timer {
  constructor(callback, interval) {
    this.timerId = null;
    this.callback = callback;
    this.interval = interval;
  }
  start() {
    if (!this.timerId) this.timerId = setInterval(this.callback, this.interval);
  }
  stop() {
    if (this.timerId) {
      clearInterval(this.timerId);
      this.timerId = null;
    }
  }
}

const timer = new Timer(() => console.log("tick every 3 seconds"), 3000);
timer.start();
setTimeout(() => {
  timer.stop();
  console.log("stop");
}, 10000);
```
