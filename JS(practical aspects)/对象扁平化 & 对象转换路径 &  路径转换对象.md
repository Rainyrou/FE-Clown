对象扁平化：

```js
const flattenObj = (target, prefix = "") => {
  const obj = {};
  for (const key in target) {
    if (target.hasOwnProperty(key)) {
      const newKey = prefix
        ? Array.isArray(target)
          ? `${prefix}[${key}]`
          : `${prefix}.${key}`
        : key;
      if (typeof target[key] === "object" && target[key] !== null)
        Object.assign(obj, flattenObj(target[key], newKey));
      else obj[newKey] = target[key];
    }
  }
  return obj;
};

const obj1 = {
  a: { a1: "vaa", b1: "bb" },
  b: ["b1", { ac: 2 }],
  c: "c1",
};

const obj2 = {
  a: "hello",
  b: 111,
  c: {
    d: "world",
    e: [1, 2, 3],
  },
};

console.log(flattenObj(obj1));
console.log(flattenObj(obj2));
```

  'c.d': 'world',

  'c.e[1]': 2,
  'c.e[2]': 3
}

const convert = (paths) => {
  const obj = {};
  for (const path of paths) {
    const parts = path.split("/").filter(Boolean);
    let cur = obj;
    for (let i = 0; i < parts.length; ++i) {
      const isLast = i === parts.length - 1;
      if (isLast) {
        cur[parts[i]] = {};
      } else {
        if (!cur[parts[i]]) cur[parts[i]] = {};
        cur = cur[parts[i]];
      }
    }
  }
  return obj;
};

const paths = [
  "/user/docs/Letter.txt",
  "/user/photos/trip.jpg",
  "/user/photos/2024/japan.png",
  "/var/log/system.log",
];

console.log(JSON.stringify(convert(paths), null, 2));
