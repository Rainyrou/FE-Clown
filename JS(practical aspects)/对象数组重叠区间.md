function isOverlap(intervals) {
  intervals.sort((a, b) => a.s - b.s);
  for (let i = 0; i < intervals.length - 1; ++i) {
    if (intervals[i].e > intervals[i + 1].s) {
      return true;
    }
  }
  return false;
}

const intervals = [
  { s: 1, e: 3 },
  { s: 2, e: 4 },
  { s: 3, e: 6 },
];

console.log(isOverlap(intervals)); // true
