```js
const isObjEmpty = (obj) => {
  if (typeof obj !== "object" || obj === null) return false;
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      if (
        typeof obj[key] === "object" &&
        obj[key] !== null &&
        !isObjEmpty(obj[key])
      )
        return false;
      else if (obj[key] !== null && obj[key] !== undefined && obj[key] !== "")
        return false;
    }
  }
  return true;
};
```

对象深度比较：

* 深比较：递归比较 JavaScript 对象所有层级的类型和属性值，处理 TypedArray、Date、RegExp、Map、Set 和循环引用等
* JSON.stringify：比较可序列化的 JavaScript 对象
* Lodash.isEqual

```js
const deepEqual = (obj1, obj2) => {
  if (obj1 === obj2) return true;
  if (
    typeof obj1 !== "object" ||
    obj1 === null ||
    typeof obj2 !== "object" ||
    obj2 === null
  )
    return false;
  const key1 = Object.keys(obj1),
    key2 = Object.keys(obj2);
  if (key1.length !== key2.length) return false;
  for (const key of key1) {
    if (!key2.includes(key) || !deepEqual(obj1[key], obj2[key])) return false;
  }
  return true;
};

const obj1 = { a: 1, b: { c: 2 } };
const obj2 = { a: 1, b: { c: 2 } };
const obj3 = { a: 1, b: { c: 3 } };
const obj4 = { a: 1, b: { c: 3, d: 4 } };
const obj5 = 5;

console.log(deepEqual(obj1, obj2)); // true
console.log(deepEqual(obj1, obj3)); // false
console.log(deepEqual(obj3, obj4)); // false
console.log(deepEqual(obj1, obj5)); // false
console.log(deepEqual(obj1, null)); // false
console.log(deepEqual(null, null)); // true
console.log(deepEqual(obj1, { a: 1, b: { c: 2 }, e: undefined })); // false
```

对象合并：

* 同名属性都为对象 → 递归合并
* 同名属性不都为对象 → 通过 `obj2` 属性值直接覆盖 `obj1` 属性值
* 惰性深拷贝：初始浅拷贝 `obj1`，在需合并嵌套对象时才递归调用 `merge` 进行深拷贝和合并

```js
const isObject = (obj) =>
  Object.prototype.toString.call(obj) === "[object Object]";

const merge = (obj1, obj2) => {
  const ans = { ...obj1 };
  for (const key in obj2) {
    ans[key] =
      isObject(ans[key]) && isObject(obj2[key])
        ? merge(ans[key], obj2[key])
        : obj2[key];
  }
  return ans;
};

const a = { b: { c: { d: { e: 1 } } } };
const x = { y: { z: 2, t: 521 } };
console.log(JSON.stringify(merge(a, x), null, 2));
```
