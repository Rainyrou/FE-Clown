const getObjPath = (obj, parentPath = []) => {
  const ans = [];
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      const curPath = [...parentPath, key];
      if (
        typeof obj[key] === "object" &&
        obj[key] !== null &&
        Object.keys(obj[key]).length > 0
      )
        ans.push(...getObjPath(obj[key], curPath));
      else ans.push(curPath);
    }
  }
  return ans;
};

const object = {
  a: { b: { c: {} } },
  d: { e: {} },
};

console.log(getObjPath(object));
