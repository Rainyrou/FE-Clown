```js
const formatTime = (dataStr, timeZone) => {
  const date = new Date(dataStr);
  const convert = new Intl.DateTimeFormat("en-US", {
    timeZone,
    year: "numeric",
    month: "2-digit",
    day: "2-digit",
    hour: "2-digit",
    minute: "2-digit",
    second: "2-digit",
    hour12: false,
  });
  const formatDate = convert.format(date);
  const [month, day, year, hour, minute, second] = formatDate.match(/\d+/g);
  return `${year}-${month}-${day} ${hour}-${minute}-${second}`;
};

const dataStr = "2024-05-17T12:00:00Z"; // UTC 时间
const timeZone = "America/New_York"; // 目标时区
console.log(formatTime(dataStr, timeZone)); // 2024-05-17 08-00-00
```
