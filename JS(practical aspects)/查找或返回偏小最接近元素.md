查找元素是否在数组中，有则返回该元素，否则返回偏小的最接近元素

```js
const find = (nums, target) => {
  if (nums.length === 0) return null;
  nums.sort((a, b) => a - b);
  let low = 0,
    high = nums.length - 1,
    ans = null;
  while (low <= high) {
    const mid = Math.floor((low + high) / 2),
      midVal = nums[mid];
    if (midVal === target) {
      return target;
    } else if (midVal < target) {
      ans = midVal;
      low = mid + 1;
    } else {
      high = mid - 1;
    }
  }
  return ans;
}

const nums = [1, 3, 5, 7, 9, 11, 13, 15];
const num1 = find(nums, 9);
console.log(num1); // 9
const num2 = find(nums, 8);
console.log(num2); // 7
const num3 = find(nums, 0);
console.log(num3); // null
```
