1. promise

```JavaScript
const red = () => console.log("red");
const green = () => console.log("green");
const yellow = () => console.log("yellow");

const light = (callback, timer) =>
  new Promise((resolve) =>
    setTimeout(() => {
      callback();
      resolve();
    }, timer)
  );

const step = () =>
  Promise.resolve()
    .then(() => light(red, 3000))
    .then(() => light(green, 2000))
    .then(() => light(yellow, 1000))
    .then(() => step());

step();
```

2. async/await

```JavaScript
const red = () => console.log("red");
const green = () => console.log("green");
const yellow = () => console.log("yellow");

const light = (callback, timer) =>
  new Promise((resolve) =>
    setTimeout(() => {
      callback();
      resolve();
    }, timer)
  );

const step = async () => {
  await light(red, 3000);
  await light(yellow, 2000);
  await light(green, 1000);
  step();
};

step();
```
