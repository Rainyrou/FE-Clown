```js
const timer = (time, num) =>
  new Promise((resolve) =>
    setTimeout(() => {
      console.log(num);
      resolve();
    }, time)
  );

const sequentLog = async () => {
  await timer(1000, 111);
  await timer(2000, 222);
  await timer(3000, 333);
};

sequentLog();
```
