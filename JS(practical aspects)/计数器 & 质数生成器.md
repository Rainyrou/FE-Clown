计数器：

```js
function Counter() {
  let count = 0;
  return {
    increment: function () {
      ++count;
      return count;
    },
    decrement: function () {
      --count;
      return count;
    },
    reset: function () {
      count = 0;
      return count;
    },
    getCount: function () {
      return count;
    },
  };
}

const counter = Counter();
console.log(counter.increment()); // 1
console.log(counter.increment()); // 2
console.log(counter.getCount()); // 2
console.log(counter.decrement()); // 1
console.log(counter.reset()); // 0
```

质数生成器：

只在首次执行内部函数时初始化变量，后续执行内部函数引用同一变量，即便外部函数执行完毕

```js
function createPrimeGenerator() {
  let cur = 1;
  return function () {
    ++cur;
    while (!isPrime(cur)) ++cur;
    return cur;
  };
}

const isPrime = (x) => {
  if (x < 2) return false;
  for (let i = 2; i * i <= x; ++i) {
    if (x % i === 0) return false;
  }
  return true;
};

const nextPrime = createPrimeGenerator();
console.log(nextPrime()); // 2
console.log(nextPrime()); // 3
console.log(nextPrime()); // 5
console.log(nextPrime()); // 7
```
