```js
function measureTime(fn) {
  return async function (...args) {
    const start = performance.now();
    const result = await fn(...args);
    const end = performance.now();
    const time = end - start;
    console.log(`execute ${time} ms`);
    return result;
  };
}

async function exampleAsyncFunction(x, y) {
  return new Promise((resolve) => setTimeout(() => resolve(x + y), 1000));
}

const timedAsyncFunction = measureTime(exampleAsyncFunction);
timedAsyncFunction(5, 10).then((res) => console.log(`result: ${res}`));
```
