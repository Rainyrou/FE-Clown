const requestWithRetry = async (fn, interval = 1000, maxCount = 3) => {
  let count = 0;
  const execute = async () => {
    try {
      ++count;
      const result = await fn();
      return result;
    } catch (error) {
      if (count < maxCount) {
        console.warn(`${count}`, `${error}`);
        await new Promise((resolve) => setTimeout(() => resolve(), interval));
        return execute();
      } else {
        throw new Error(`${maxCount}`, `${error}`);
      }
    }
  };
  return execute();
};

const simulateRequest = () =>
  new Promise((resolve, reject) =>
    setTimeout(() => resolve("Request succeeded!"), 500)
  );

requestWithRetry(simulateRequest, 1000, 3)
  .then((result) => console.log(result)) // Request succeeded!
  .catch((error) => console.error(error.message));
