- `unit`：基础权位，当前的小单位倍率如 1/10/100/1000），默认为 1
- `section`：分节权位，当前的大区域倍率如 1/10000/100000000，默认是 1
* 核心公式：

$$当前数字的实际值 = 数字 \times unit \times section$$

```js
const Chinese2Int = (chineseNumber) => {
  const numberMap = {
    零: 0,
    一: 1,
    二: 2,
    三: 3,
    四: 4,
    五: 5,
    六: 6,
    七: 7,
    八: 8,
    九: 9,
  };

  const unitMap = {
    十: 10,
    百: 100,
    千: 1000,
    万: 10000,
    亿: 100000000,
  };

  let ans = 0,
    unit = 1,
    section = 1;
  for (let i = chineseNumber.length - 1; i >= 0; --i) {
    const char = chineseNumber[i];
    if (char in unitMap) {
      const val = unitMap[char];
      if (val >= 10000) { // 判断是否为万/亿
        if (val > section) { // 在亿万中遍历至亿，更新分节权位并重置基础权位
          section = val; 
          unit = 1;
        } else {
          unit = val; // 在万亿中遍历至万，此时万不为分节符，而为修饰亿的数量单位
        }
      } else {
        unit = val;
      }
    } else if (char in numberMap) {
      const num = numberMap[char];
      ans += num * unit * section;
    }
  }
  if (chineseNumber[0] === "十") ans += 1 * 10 * section; // 兼容以十开头省略前面的一之口语情况
  return ans;
};

console.log(Chinese2Int("一万亿")); // 1000000000000
console.log(Chinese2Int("一千二百三十四")); // 1234
console.log(Chinese2Int("十二")); // 12
console.log(Chinese2Int("一万零三十")); // 10030
console.log(Chinese2Int("一亿零二万五千三百")); // 100025300
console.log(Chinese2Int("三千五百二十")); // 3520
console.log(Chinese2Int("十万")); // 100000
```
