```js
function myCalculator(num) {
  return {
    value: num,
    add: function (x) {
      this.value += x;
      return this;
    },
    minus: function (x) {
      this.value -= x;
      return this;
    },
    multi: function (x) {
      this.value *= x;
      return this;
    },
    div: function (x) {
      this.value /= x;
      return this;
    },
  };
}

console.log(myCalculator(1).add(4).minus(2).multi(3).div(3).value); // 3
```
