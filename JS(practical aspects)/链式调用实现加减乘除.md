```js
function myCalculator(num) {
  return {
    value: num,
    add(x) {
      this.value += x;
      return this;
    },
    minus(x) {
      this.value -= x;
      return this;
    },
    multi(x) {
      this.value *= x;
      return this;
    },
    div(x) {
      this.value /= x;
      return this;
    }
  };
}

console.log(myCalculator(1).add(4).minus(2).multi(3).div(3).value); // 3
```
