```js
const logWithDelay = (num, delay) =>
  new Promise((resolve) =>
    setTimeout(() => {
      console.log(num);
      resolve();
    }, delay * 1000)
  );

const execute = async (x = 1, y) => {
  for (let i = 1; i <= y; ++i) await logWithDelay(i, x);
};

execute(1, 10);
```

间隔重复执行函数：

```js
function repeat(func, times, wait) {
  return function (...args) {
    let count = 0;
    const execute = () => {
      func.apply(this, args);
      ++count;
      if (count < times) setTimeout(execute, wait);
    };
    execute();
  };
}

const repeatFunc = repeat(alert, 4, 3000);
repeatFunc("hellworld");
```
