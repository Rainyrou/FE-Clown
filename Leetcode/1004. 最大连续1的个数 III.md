[1004. 最大连续1的个数 III](https://leetcode.cn/problems/max-consecutive-ones-iii/)

```javascript
var longestOnes = function (nums, k) {
  let left = 0,
    right = 0,
    maxLen = 0,
    zero = 0;
  while (right < nums.length) {
    if (nums[right] === 0) ++zero;
    while (zero > k) {
      if (nums[left] === 0) --zero;
      ++left;
    }
    maxLen = Math.max(maxLen, right - left + 1);
    ++right;
  }
  return maxLen;
};
```
