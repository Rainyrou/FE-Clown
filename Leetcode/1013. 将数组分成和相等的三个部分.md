[1013. 将数组分成和相等的三个部分](https://leetcode.cn/problems/partition-array-into-three-parts-with-equal-sum/)

变式：返回下标

```js
const find = (arr) => {
  const sum = arr.reduce((a, b) => a + b, 0);
  if (sum % 3) return [-1, -1];
  const target = sum / 3;
  let curSum = 0,
    count = 0,
    x = -1,
    y = -1;
  for (let i = 0; i < arr.length; ++i) {
    curSum += arr[i];
    if (curSum === target) {
      ++count;
      if (count === 1) {
        x = i;
        curSum = 0;
      } else if (count === 2) {
        if (i < arr.length - 1) {
          y = i;
          return [x, y];
        }
      }
    }
  }
  return [-1, -1];
};
```
