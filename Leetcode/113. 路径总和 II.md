[113. 路径总和 II](https://leetcode.cn/problems/path-sum-ii/description/)

```javascript
var pathSum = function(root, targetSum) {
  if(!root) return [];
  const ans = [], path = [root.val];
  const backTracking = (node, target) => {
    if(!node.left && !node.right && !target) {
      ans.push([...path]);
      return;
    }
    if(node.left) {
      path.push(node.left.val);
      backTracking(node.left, target - node.left.val);
      path.pop();
    }
    if(node.right) {
      path.push(node.right.val);
      backTracking(node.right, target - node.right.val);
      path.pop();
    }
  }
  backTracking(root, targetSum - root.val);
  return ans;
};
```

```javascript
var pathSum = function (root, targetSum) {
  if (!root) return [];
  const ans = [],
    path = [];
  const backTracking = (node, target) => {
    path.push(node.val);
    target -= node.val;
    if (!node.left && !node.right && !target) {
      ans.push([...path]);
      path.pop();
      return;
    }
    if (node.left) backTracking(node.left, target);
    if (node.right) backTracking(node.right, target);
    path.pop();
  };
  backTracking(root, targetSum);
  return ans;
};
```
