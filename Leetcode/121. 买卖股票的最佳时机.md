###### 暴力破解（201 / 212 个通过的测试用例）

```javascript
var maxProfit = function(prices) {
  let maxSum = 0;
  for(let i = 0;i < prices.length;++i) {
    for(let j = i + 1;j < prices.length;++j) {
      maxSum = Math.max(maxSum, prices[j] - prices[i]);
    }
  }
  return maxSum;
};
```

###### 贪心

```javascript
var maxProfit = function(prices) {
  let maxSum = 0, lowest = Infinity;
  for(let i = 0;i < prices.length;++i) {
    lowest = Math.min(lowest, prices[i]);
    maxSum = Math.max(maxSum, prices[i] - lowest);
  }
  return maxSum;
};
```

###### 动态规划

```javascript
var maxProfit = function (prices) {
  const len = prices.length;
  const dp = new Array(len).fill(0).map(() => new Array(2).fill(0));
  dp[0][1] = -prices[0];
  for (let i = 1; i < len; ++i) {
    dp[i][0] = Math.max(dp[i - 1][0], dp[i - 1][1] + prices[i]);
    dp[i][1] = Math.max(dp[i - 1][1], -prices[i]);
  }
  return dp[len - 1][0] < 0 ? 0 : dp[len - 1][0];
};
```
