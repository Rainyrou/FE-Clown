[13. 罗马数字转整数](https://leetcode.cn/problems/roman-to-integer/)

```javascript
var romanToInt = function (s) {
  let ans = 0;
  const map = new Map();
  map.set("I", 1);
  map.set("V", 5);
  map.set("X", 10);
  map.set("L", 50);
  map.set("C", 100);
  map.set("D", 500);
  map.set("M", 1000);
  for (let i = 0; i < s.length; ++i) {
    const val = map.get(s[i]);
    if (i < s.length - 1 && val < map.get(s[i + 1])) ans -= val;
    else ans += val;
  }
  return ans;
};
```
