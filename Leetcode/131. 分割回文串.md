var partition = function(s) {
  const ans = [];
  const backTracking = (i, path) => {
    if(i === s.length) {
      ans.push(path.slice());
      return;
    }
    for(let j = i;j < s.length;++j) {
      if(!isPalindrome(s, i, j)) continue;
      path.push(s.slice(i, j + 1));
      backTracking(j + 1, path);
      path.pop();
    }
  }
  backTracking(0, []);
  return ans;
};

const isPalindrome = (s, left, right) => {
  for(let i = left, j = right;i < j;++i, --j) if(s[i] !== s[j]) return false;
  return true;
}
