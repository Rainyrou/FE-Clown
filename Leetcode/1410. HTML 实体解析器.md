[1410. HTML 实体解析器](https://leetcode.cn/problems/html-entity-parser/description/)

```javascript
var entityParser = function (text) {
  const map = [
    [/&quot;/g, '"'],
    [/&apos;/g, "'"],
    [/&gt;/g, ">"],
    [/&lt;/g, "<"],
    [/&frasl;/g, "/"],
    [/&amp;/g, "&"], // &amp; 为实体前缀，应该最后解析
  ];
  for (const [key, value] of map) text = text.replace(key, value);
  return text;
};
```
