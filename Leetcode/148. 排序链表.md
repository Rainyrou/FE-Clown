[排序链表](https://leetcode.cn/problems/sort-list/?envType=study-plan-v2&envId=top-100-liked)

###### 暴力破解

```javascript
var sortList = function(head) {
  if(!head || !head.next) return head;
  const ans = [];
  while(head) {
    ans.push(head.val);
    head = head.next;
  }
  ans.sort((a, b) => a - b);
  let cur = new ListNode(ans.shift(), null), p = cur;
  while(ans.length) {
    let next = new ListNode(ans.shift(), null);
    cur.next = next;
    cur = cur.next;
  }
  return p;
};
```

###### 割断 + 连接

```javascript
var sortList = function(head) {
  if(!head || !head.next) return head;
  const ans = [];
  while(head) {
    const temp = head.next;
    head.next = null;
    ans.push(head);
    head = temp;
  }
  ans.sort((a, b) => a.val - b.val).reduce((i, j) => i.next = j);
  return ans[0];
};
```

###### 快慢指针 + 递归 + 归并排序

```javascript
var sortList = function (head) {
  if (!head || !head.next) return head;
  const mid = middleNode(head);
  const left = sortList(head),
    right = sortList(mid);
  return merge(left, right);
};

const middleNode = (head) => {
  let dummyHead = new ListNode(-1, head),
    slow = dummyHead,
    fast = dummyHead;
  while (fast.next && fast.next.next) {
    slow = slow.next;
    fast = fast.next.next;
  }
  const cur = slow.next;
  slow.next = null;
  return cur;
};

const merge = (l1, l2) => {
  let dummyHead = new ListNode(-1),
    cur = dummyHead;
  while (l1 && l2) {
    if (l1.val < l2.val) {
      cur.next = l1;
      l1 = l1.next;
    } else {
      cur.next = l2;
      l2 = l2.next;
    }
    cur = cur.next;
  }
  cur.next = l1 || l2;
  return dummyHead.next;
};
```
