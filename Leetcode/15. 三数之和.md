[三数之和](https://leetcode.cn/problems/3sum/)

n数之和：排序数组，递归降维，将n数之和转化为 n-1 数之和，直到降为两数之和，再通过双指针求解
时间复杂度：O(n^(n-1)) -> 各层递归均遍历数组O(n)，从n到2经过n-2层循环，当n=2时，双指针遍历O(n)
空间复杂度：O(n) -> 原地排序O(1)，递归调用栈深度为 O(n)
n 过大的溢出问题：BigInt + 优化剪枝（在递归过程中，提前终止不可能的情况）


```js
const findNSum = (nums, n, target) => {
  const ans = [];
  nums.sort((a, b) => a - b);
  if (n < 2 || nums.length < n) return ans;
  if (n === 2) {
    let left = 0,
      right = nums.length - 1;
    while (left < right) {
      const sum = nums[left] + nums[right];
      if (sum === target) {
        ans.push([nums[left], nums[right]]);
        while (left < right && nums[left] === nums[left + 1]) ++left;
        ++left;
        while (left < right && nums[right] === nums[right - 1]) --right;
        --right;
      } else if (sum < target) {
        ++left;
      } else {
        --right;
      }
    }
    return ans;
  }
  for (let i = 0; i < nums.length - n + 1; ++i) {
    if (i > 0 && nums[i] === nums[i - 1]) continue;
    const subProblems = findNSum(nums.slice(i + 1), n - 1, target - nums[i]);
    for (const subProblem of subProblems) ans.push([nums[i], ...subProblem]);
  }
  return ans;
};

console.log(findNSum([1, 0, -1, 0, -2, 2], 4, 0)); // [ [ -2, -1, 1, 2 ], [ -2, 0, 0, 2 ], [ -1, 0, 0, 1 ] ]
```

###### 暴力破解

```JavaScript
var threeSum = function(nums) {
    const ans = [];
    const set = new Set();
    for(let i = 0;i < nums.length;++i) {
        for(let j = i + 1;j < nums.length;++j) {
            for(let l = j + 1;l < nums.length;++l) {
                if(nums[i] + nums[j] + nums[l] === 0) {
                    const sorted = [nums[i], nums[j], nums[l]].sort((a, b) => a - b).toString();
                    if(!set.has(sorted)) {
                        set.add(sorted);
                        ans.push([nums[i], nums[j], nums[l]]);
                    }
                }
            }
        }
    }
    return ans;
};
```

###### 排序 + 双指针 + 去重

```javascript
var threeSum = function(nums) {
  nums.sort((a, b) => a - b);
  const ans = [];
  for(let i = 0;i < nums.length;++i) {
    if(i > 0 && nums[i - 1] === nums[i]) continue;
    if(nums[i] > 0) break;
    let left = i + 1, right = nums.length - 1;
    while(left < right) {
      let sum = nums[i] + nums[left] + nums[right];  
      if(sum === 0) {
        ans.push([nums[i], nums[left], nums[right]]);
        while(left < right && nums[left] === nums[left + 1]) ++left;
        ++left;
        while(left < right && nums[right] === nums[right - 1]) --right;
        --right;
      } else if(sum < 0) {
        ++left;
      } else {
        --right;
      }
    }
  }
  return ans;
};
```
