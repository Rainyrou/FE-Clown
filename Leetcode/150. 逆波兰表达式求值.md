[150. 逆波兰表达式求值](https://leetcode.cn/problems/evaluate-reverse-polish-notation/description/)

```javascript
var evalRPN = function (tokens) {
  const stk = [];
  const map = new Map([
    ["+", (a, b) => b + a],
    ["-", (a, b) => b - a],
    ["*", (a, b) => b * a],
    ["/", (a, b) => Math.trunc(b / a)],
  ]);
  for (const token of tokens) {
    if (!map.has(token)) stk.push(Number(token));
    else stk.push(map.get(token)(stk.pop(), stk.pop()));
  }
  return stk.pop();
};
```
