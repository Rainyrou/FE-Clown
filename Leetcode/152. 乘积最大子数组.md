[152. 乘积最大子数组](https://leetcode.cn/problems/maximum-product-subarray/description/?envType=study-plan-v2&envId=top-100-liked)

```javascript
var maxProduct = function (nums) {
  let curMax = 1,
    curMin = 1,
    ans = -Infinity;
  for (const num of nums) {
    let temp1 = curMax * num,
      temp2 = curMin * num;
    // 更新当前的最小值和最大值
    curMax = Math.max(temp1, temp2, num);
    curMin = Math.min(temp1, temp2, num);
    ans = Math.max(curMax, ans); // 更新全局的最大值
  }
  return ans;
};
```
