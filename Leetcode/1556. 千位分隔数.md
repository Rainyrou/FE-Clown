[1556. 千位分隔数](https://leetcode.cn/problems/thousand-separator/description/)

```javascript
var thousandSeparator = function (n) {
  const reversed = n.toString().split("").reverse();
  const chunck = [];
  for (let i = 0; i < reversed.length; i += 3)
    chunck.push(reversed.slice(i, i + 3).join(""));
  const formatted = chunck.join(".").split("").reverse().join("");
  return formatted;
};
```
