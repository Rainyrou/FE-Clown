```javascript
var findRepeatedDnaSequences = function (s) {
  const ans = [],
    map = new Map();
  for (let i = 0; i <= s.length - 10; ++i) {
    const str = s.slice(i, i + 10);
    map.set(str, (map.get(str) || 0) + 1);
    if (map.get(str) === 2) ans.push(str);
  }
  return ans;
};
```

```javascript
var findRepeatedDnaSequences = function (s) {
  const ans = new Set(),
    set = new Set();
  for (let i = 0; i <= s.length - 10; ++i) {
    const str = s.slice(i, i + 10);
    if (set.has(str)) ans.add(str);
    else set.add(str);
  }
  return [...ans];
};
```
