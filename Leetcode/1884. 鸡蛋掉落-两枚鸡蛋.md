[1884. 鸡蛋掉落-两枚鸡蛋](https://leetcode.cn/problems/egg-drop-with-2-eggs-and-n-floors/description/)

```javascript
var twoEggDrop = function(n) {
  const dp = new Array(n + 1).fill(Infinity);
  dp[0] = 0;
  for(let i = 1;i <= n;++i) {
    for(let j = 1;j <= i;++j) {
      dp[i] = Math.min(dp[i], Math.max(j - 1, dp[i - j]) + 1);
    } 
  }
  return dp[n];
};
```

###### 数学规律

```javascript
var twoEggDrop = function(n) {
  let sum = 0, i = 1;
  for(i = 1;i < n;++i) {
    sum += i;
    if(sum >= n) break;
  }
  return i;
};
```
