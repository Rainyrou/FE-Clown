###### 正则

```javascript
var isValid = function(s) {
  while(s.length) {
    const temp = s;
    s = s.replace('()', '');
    s = s.replace('[]', '');
    s = s.replace('{}', '');
    if(temp === s) return false;
  }
  return true;
};
```

###### 模拟栈

```javascript
var isValid = function(s) {
  if(s.length % 2) return false;
  const stk = [];
  const map = new Map([
    [')', '('],
    [']', '['],
    ['}', '{']
  ])
  for(const ch of s) {
    if(map.has(ch)) {
      if(!stk.length || stk[stk.length - 1] !== map.get(ch)) return false;
      stk.pop();
    } else {
      stk.push(ch);
    }
  }
  return !stk.length;
};
```

栈只存左符号
