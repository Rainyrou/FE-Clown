	[有效的括号](https://leetcode.cn/problems/valid-parentheses/?envType=study-plan-v2&envId=top-100-liked)

###### 正则

```javascript
var isValid = function(s) {
  while(s.length) {
    const temp = s;
    s = s.replace('()', '');
    s = s.replace('[]', '');
    s = s.replace('{}', '');
    if(temp === s) return false;
  }
  return true;
};
```

###### 模拟栈

```javascript
var isValid = function (s) {
  if(s.length % 2) return false;
  const map = new Map([
    [")", "("],
    ["]", "["],
    ["}", "{"],
  ]);
  const stk = []; // 栈只存左符号
  for (const ch of s) {
    if (map.has(ch)) {
      if (!stk.length || map.get(ch) !== stk[stk.length - 1]) return false;
      stk.pop();
    } else {
      stk.push(ch);
    }
  }
  return !stk.length;
};
```
