var minSubArrayLen = function (target, nums) {
  let minLen = Infinity,
    left = 0,
    right = 0,
    curSum = 0;
  while (right < nums.length) {
    curSum += nums[right];
    while (curSum >= target) {
      minLen = Math.min(minLen, right - left + 1);
      curSum -= nums[left];
      ++left;
    }
    ++right;
  }
  return minLen === Infinity ? 0 : minLen;
}
