[228. 汇总区间](https://leetcode.cn/problems/summary-ranges/description/)

```javascript
var summaryRanges = function (nums) {
  const ans = [],
    len = nums.length;
  let i = 0;
  while (i < len) {
    let cur = i;
    while (i + 1 < len && nums[i] === nums[i + 1] - 1) ++i;
    if (cur !== i) ans.push(`${nums[cur]}->${nums[i]}`);
    else ans.push(`${nums[cur]}`);
    ++i;
  }
  return ans;
};
```
