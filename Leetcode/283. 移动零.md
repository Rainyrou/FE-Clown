[移动零](https://leetcode.cn/problems/move-zeroes/description/?envType=study-plan-v2&envId=top-100-liked)

###### splice + push

```javascript
var moveZeroes = function (nums) {
  let len = nums.length;
  for (let i = 0; i < len; ++i) {
    if (nums[i] === 0) {
      nums.splice(i, 1);
      nums.push(0);
      --i;
      --len;
    }
  }
};
```

###### 快慢指针

```js
var moveZeroes = function(nums) {
  let slow = 0;
  for(let fast = 0;fast < nums.length;++fast) {
    if(nums[fast]) {
      [nums[fast], nums[slow]] = [nums[slow], nums[fast]];
      ++slow;
    }
  }
  return nums;
};
```
