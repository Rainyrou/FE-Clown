[328. 奇偶链表](https://leetcode.cn/problems/odd-even-linked-list/description/)

```javascript
var oddEvenList = function (head) {
  if (!head || !head.next) return head;
  let odd = head,
    even = head.next,
    evenHead = even;
  while (even && even.next) {
    odd.next = even.next;
    odd = odd.next;
    even.next = odd.next;
    even = even.next;
  }
  odd.next = evenHead;
  return head;
};
```
