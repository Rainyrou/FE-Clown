动态规划：

```javascript
var combinationSum = function (candidates, target) {
  const dp = new Array(target + 1).fill(0).map(() => []);
  dp[0] = [[]];
  for (let i = 0; i < candidates.length; ++i) {
    for (let j = candidates[i]; j <= target; ++j) {
      for (const item of dp[j - candidates[i]])
        dp[j].push([...item, candidates[i]]);
    }
  }
  return dp[target];
};
```

回溯：

```js
const combinationSum = (candidates, target) => {
  candidates.sort((a, b) => a - b);
  const ans = [],
    path = [];
  const backTracking = (i, curSum) => {
    if (curSum === target) {
      ans.push(path.slice());
      return;
    }
    for (let j = i; j < candidates.length; ++j) {
      const val = candidates[j];
      if (val > target - curSum) return;
      path.push(val);
      curSum += val;
      backTracking(j, curSum);
      path.pop();
      curSum -= val;
    }
  };
  backTracking(0, 0);
  return ans;
};
```
