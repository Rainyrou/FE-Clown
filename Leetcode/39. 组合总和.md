var combinationSum = function(candidates, target) {
  candidates.sort((a, b) => a - b);
  const ans = [];
  const backTracking = (i, curSum, path) => {
    if(curSum === target) {
      ans.push(path.slice());
      return;
    }
    for(let j = i;j < candidates.length;++j) {
      const val = candidates[j];
      if(val > target - curSum) return;
      path.push(val);
      curSum += val;
      backTracking(j, curSum, path);
      path.pop();
      curSum -= val;
    }
  }
  backTracking(0, 0, []);
  return ans;
};
