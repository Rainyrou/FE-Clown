[43. 字符串相乘](https://leetcode.cn/problems/multiply-strings/description/)

```javascript
var multiply = function (num1, num2) {
  const m = num1.length,
    n = num2.length;
  const nums = new Array(m + n).fill(0);
  for (let i = m - 1; i >= 0; --i) {
    for (let j = n - 1; j >= 0; --j) {
      const sum = num1[i] * num2[j] + nums[i + j + 1];
      nums[i + j] += Math.floor(sum / 10);
      nums[i + j + 1] = sum % 10;
    }
  }
  let ans = nums.join("");
  while (ans[0] === "0") ans = ans.slice(1);
  return ans.length ? ans : '0';
};
```
