[445. 两数相加 II](https://leetcode.cn/problems/add-two-numbers-ii/description/)

```javascript
var addTwoNumbers = function (l1, l2) {
  l1 = reverseList(l1);
  l2 = reverseList(l2);
  const l3 = add(l1, l2);
  return reverseList(l3);
};

const add = (l1, l2) => {
  let dummyHead = new ListNode(-1),
    cur = dummyHead,
    carry = 0;
  while (l1 || l2 || carry) {
    const x = l1 ? l1.val : 0,
      y = l2 ? l2.val : 0;
    const sum = x + y + carry;
    cur.next = new ListNode(sum % 10);
    carry = Math.floor(sum / 10);
    cur = cur.next;
    if (l1) l1 = l1.next;
    if (l2) l2 = l2.next;
  }
  return dummyHead.next;
};

const reverseList = (head) => {
  let cur = head,
    pre = null,
    next = null;
  while (cur) {
    next = cur.next;
    cur.next = pre;
    pre = cur;
    cur = next;
  }
  return pre;
};
```
