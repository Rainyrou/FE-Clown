var permute = function(nums) {
  const ans = [], visited = new Array(nums.length).fill(false);
  const dfs = path => {
    if(path.length === nums.length) {
      ans.push(path.slice()); // ans.push([...path]) or ans.push(Array.from(path));
      return;
    }
    for(let i = 0;i < nums.length;++i) {
      if(visited[i]) continue;
      path.push(nums[i]);
      visited[i] = true;
      dfs(path);
      path.pop();
      visited[i] = false;
    }
  }
  dfs([]);
  return ans;
};

const permute = (nums) => {
  const ans = [];
  const backTracking = (path) => {
    if (path.length === nums.length) {
      ans.push(path.slice());
      return;
    }
    for (let i = 0; i < nums.length; ++i) {
      const index = path.indexOf(nums[i]);
      if (index !== -1) continue;
      path.push(nums[i]);
      backTracking(path);
      path.pop();
    }
  };
  backTracking([]);
  return ans;
};

var permute = function(nums) {
  const res = [];
  const path = [];
  const dfs = remain => {
    if(remain.length === 0) {
      res.push([...path]);
      return;
    }
    for(let i = 0;i < remain.length;++i) {
      path.push(remain[i]);
      dfs([...remain.slice(0, i), ...remain.slice(i + 1)]);
      path.pop();
    }
  }
  dfs(nums);
  return res;
};
