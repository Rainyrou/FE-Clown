[468. 验证IP地址](https://leetcode.cn/problems/validate-ip-address/)

```javascript
var validIPAddress = function (queryIP) {
  const arr4 = queryIP.split("."),
    arr6 = queryIP.split(":");
  if (
    arr4.length === 4 &&
    arr4.every((item) => /^0$|^([1-9]\d{0,2})$/.test(item) && item < 256)
  )
    return "IPv4";
  if (
    arr6.length === 8 &&
    arr6.every((item) => /^[0-9a-fA-F]{1,4}$/.test(item))
  )
    return "IPv6";
  return "Neither";
};
```
