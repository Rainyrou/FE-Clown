[485. 最大连续 1 的个数](https://leetcode.cn/problems/max-consecutive-ones/description/)

```javascript
var findMaxConsecutiveOnes = function (nums) {
  let i = 0,
    maxLen = 0;
  while (i < nums.length) {
    if (nums[i] === 1) {
      let len = 1;
      while (nums[++i] === 1) ++len;
      maxLen = Math.max(maxLen, len);
    }
    ++i;
  }
  return maxLen;
};
```
