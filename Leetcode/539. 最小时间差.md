[539. 最小时间差](https://leetcode.cn/problems/minimum-time-difference/description/)

```javascript
const findMinDifference = (timePoints) => {
  const mins = timePoints
    .map((time) => {
      const [hour, min] = time.split(":").map(Number);
      return hour * 60 + min;
    })
    .sort((a, b) => a - b);
  let ans = 1440 + mins[0] - mins[timePoints.length - 1];
  for (let i = 1; i < timePoints.length; ++i)
    ans = Math.min(ans, mins[i] - mins[i - 1]);
  return ans;
};
```
