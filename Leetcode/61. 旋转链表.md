[61. 旋转链表](https://leetcode.cn/problems/rotate-list/description/)

```javascript
var rotateRight = function (head, k) {
  if (head === null || k === 0) return head;
  let len = 1;
  let tail = head;
  while (tail.next) {
    ++len;
    tail = tail.next;
  }
  k %= len;
  if (k === 0) return head;
  let newTail = head;
  for (let i = 1; i < len - k; ++i) newTail = newTail.next;
  let newHead = newTail.next;
  tail.next = head;
  newTail.next = null;
  return newHead;
};
```
