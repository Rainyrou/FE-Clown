[670. 最大交换](https://leetcode.cn/problems/maximum-swap/)

```javascript
var maximumSwap = function (num) {
  const s = num.toString();
  let index = s.length - 1;
  let q1 = -1,
    q2 = 0;
  for (let i = s.length - 2; i >= 0; --i) {
    if (s[i] > s[index]) {
      index = i;
    } else if (s[i] < s[index]) {
      q1 = i;
      q2 = index;
    }
  }
  if (q1 === -1) return num;
  const t = s.split("");
  [t[q1], t[q2]] = [t[q2], t[q1]];
  return Number(t.join(""));
};
```
