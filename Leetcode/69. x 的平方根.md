[69. x 的平方根](https://leetcode.cn/problems/sqrtx/description/)

二分查找：

```javascript
var mySqrt = function (x) {
  if (x === 0) return 0;
  let ans = 0;
  let left = 1,
    right = x;
  while (left <= right) {
    const mid = Math.floor((right - left) / 2) + left,
      cur = mid * mid;
    if (cur === x) {
      return mid;
    } else if (cur < x) {
      ans = mid;
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  return ans;
};
```

`cur < x` 时，`mid` 为有效值且可能比之前的 `ans` 大，更新 `ans` 以寻找更优解，而 `cur > x` 时，`mid` 超出题意范围，为无效值

暴力破解：

```javascript
var mySqrt = function (x) {
  let i = 0;
  while (i * i <= x) ++i;
  return i - 1;
};
```
