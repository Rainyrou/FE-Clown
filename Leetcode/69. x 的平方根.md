[69. x 的平方根](https://leetcode.cn/problems/sqrtx/description/)

二分查找：

```javascript
var mySqrt = function (x) {
  if (x === 0) return 0;
  let ans = 0;
  let left = 1,
    right = x;
  while (left <= right) {
    const mid = Math.floor((right - left) / 2) + left,
      cur = mid * mid;
    if (cur === x) {
      return mid;
    } else if (cur < x) {
      ans = mid;
      left = mid + 1;
    } else {
      right = mid - 1;
    }
  }
  return ans;
};
```

暴力破解：

```javascript
var mySqrt = function (x) {
  if(x === 0) return 0;
  let i = 0;
  while (i * i < x) {
    ++i;
    if (i * i === x) return i;
  }
  return i - 1;
};
```

```javascript
var mySqrt = function (x) {
  const ans = [];
  for (let i = 0; i * i <= x; ++i) ans.push(i);
  return ans[ans.length - 1];
};
```
