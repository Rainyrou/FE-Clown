[简化路径](https://leetcode.cn/problems/simplify-path/description/)

```javascript
var simplifyPath = function (path) {
  const arr = path.split("/").filter(Boolean),
    stk = [];
  for (const item of arr) {
    if (item === "..") {
      if (stk.length) {
        stk.pop();
      }
    } else if (item && item !== ".") {
      stk.push(item);
    }
  }
  return "/" + stk.join("/");
};
```
