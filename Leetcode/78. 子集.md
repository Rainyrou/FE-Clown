var subsets = function(nums) {
  const ans = [];
  const backTracking = (i, path) => {
    if(i === nums.length) {
      ans.push(path.slice());
      return;
    }
    path.push(nums[i]);
    backTracking(i + 1, path);
    path.pop();
    backTracking(i + 1, path);
  }
  backTracking(0, []);
  return ans;
};
