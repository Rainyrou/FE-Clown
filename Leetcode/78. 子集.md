| 维度   | 范围选择法                | 二元决策法            |
| ---- | -------------------- | ---------------- |
| 核心思路 | 从当前索引开始，选择要加入的元素     | 对各个元素，明确选或不选两种分支 |
| 记录时机 | 每次递归即记录（中间状态均为有效子集）  | 遍历完所有元素后记录       |
| 去重机制 | 通过 j 从 i 开始以避免重复选择元素 | 通过索引顺序以避免重复选择元素  |
| 本质   | 遍历所有可能的元素范围组合        | 遍历所有可能的选/不选组合    |

```javascript
var subsets = function (nums) {
  const ans = [];
  const backTracking = (i, path) => {
    ans.push(path.slice());
    if (i === nums.length) return;
    for (let j = i; j < nums.length; ++j) {
      path.push(nums[j]);
      backTracking(j + 1, path);
      path.pop();
    }
  };
  backTracking(0, []);
  return ans;
};
```

```js
const subsets = (nums) => {
  const ans = [],
    path = [];
  const backTracking = (i) => {
    if (i === nums.length) {
      ans.push(path.slice());
      return;
    }
    path.push(nums[i]);
    backTracking(i + 1);
    path.pop();
    backTracking(i + 1);
  };
  backTracking(0);
  return ans;
};
```
