var numDecodings = function(s) {
  if(s[0] === '0') return 0;
  const len = s.length, dp = [1, 1, ...new Array(len - 1).fill(0)];
  for(let i = 2;i <= len;++i) {
    const lastOne = s.slice(i - 1, i), lastTwo = s.slice(i - 2, i);
    if(0 < lastOne && lastOne < 10) dp[i] += dp[i - 1];
    if(10 <= lastTwo && lastTwo <= 26) dp[i] += dp[i - 2];
  }
  return dp[len];
};
