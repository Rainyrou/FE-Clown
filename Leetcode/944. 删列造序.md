[944. 删列造序](https://leetcode.cn/problems/delete-columns-to-make-sorted/)

```javascript
var minDeletionSize = function (strs) {
  const row = strs.length,
    col = strs[0].length;
  let ans = 0;
  for (let j = 0; j < col; ++j) {
    for (let i = 1; i < row; ++i) {
      if (strs[i - 1][j] > strs[i][j]) {
        ++ans;
        break;
      }
    }
  }
  return ans;
};
```
