[验证二叉搜索树](https://leetcode.cn/problems/validate-binary-search-tree/?envType=study-plan-v2&envId=top-100-liked)

###### 中序遍历递归

```javascript
var isValidBST = function (root) {
  if (!root) return true;
  const ans = [];
  const inOrder = (node) => {
    if (!node) return;
    inOrder(node.left);
    ans.push(node.val);
    inOrder(node.right);
  };
  inOrder(root);
  return ans.every(
    (val, index) => val < ans[index + 1] || index === ans.length - 1
  );
};
```

###### 中序遍历迭代

```javascript
var isValidBST = function(root) {
  if(!root) return true;
  const stk = [];
  let pre = -Infinity;
  while(root || stk.length) {
    while(root) {
      stk.push(root);
      root = root.left;
    }
    root = stk.pop();
    if(root.val <= pre) return false;
    pre = root.val;
    root = root.right;
  }
  return true;
};
```
