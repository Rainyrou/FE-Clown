[LCR 143. 子结构判断](https://leetcode.cn/problems/shu-de-zi-jie-gou-lcof/description/)

```javascript
var isSubStructure = function (A, B) {
  if (!A || !B) return false;
  return (
    isSameStructure(A, B) ||
    isSubStructure(A.left, B) ||
    isSubStructure(A.right, B)
  );
};

const isSameStructure = (node1, node2) => {
  if (!node2) return true;
  if (!node1 || node1.val !== node2.val) return false;
  return (
    isSameStructure(node1.left, node2.left) &&
    isSameStructure(node1.right, node2.right)
  );
};
```

外层递归 -> 以 tree1的各个节点为根节点
内层递归 -> 判断各个根节点对应的子树是否与 tree2 结构完全一致 + 节点值完全相同
