[LCR 155. 将二叉搜索树转化为排序的双向链表](https://leetcode.cn/problems/er-cha-sou-suo-shu-yu-shuang-xiang-lian-biao-lcof/description/)

```javascript
var treeToDoublyList = function (root) {
  let pre = null,
    head = null;
  if (!root) return root;
  const dfs = (root) => {
    if (!root) return null;
    dfs(root.left);
    pre !== null ? (pre.right = root) : (head = root);
    root.left = pre;
    pre = root;
    dfs(root.right);
  };
  dfs(root);
  head.left = pre;
  pre.right = head;
  return head;
};
```
