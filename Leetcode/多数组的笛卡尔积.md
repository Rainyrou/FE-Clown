const generateCombinations = (nums) => {
  const ans = [];
  const backTracking = (i, path) => {
    if (i === nums.length) {
      ans.push(path.slice());
      return;
    }
    for (const item of nums[i]) {
      path.push(item);
      backTracking(i + 1, path);
      path.pop();
    }
  };
  backTracking(0, []);
  return ans;
};

const input = [
  [1, 2, 3],
  [4, 5],
  [6, 7, 8],
];
console.log(generateCombinations(input));
