const find = (str) => {
  const map = new Map();
  for (const ch of str) map.set(ch, (map.get(ch) || 0) + 1);
  let maxCount = 0,
    ans = [];
  map.forEach((count, ch) => {
    if (count > maxCount) {
      maxCount = count;
      ans = [ch];
    } else if (count === maxCount) {
      ans.push(ch);
    }
  });
  return [ans, maxCount];
};
