JSX：

1. 只有一个 HTML 根标签
2. HTML 标签必须闭合
3. HTML 标签混入 JavaScript 表达式 -> `{}`
4. CSS类名 -> `className`
5. 内联 CSS 样式 -> `style={{key: val}}`

function + class

```HTML
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Hello React</title>
</head>
<body>
    <div id="test1"></div>
    <div id="test2"></div>
    <div id="test3"></div>
    <script type="text/javascript" src="https://unpkg.com/react@17/umd/react.development.js"></script>
    <script type="text/javascript" src="https://unpkg.com/react-dom@17/umd/react-dom.development.js"></script>
    <script type="text/javascript" src="https://unpkg.com/@babel/standalone/babel.min.js"></script>
    <script type="text/javascript" src="https://unpkg.com/prop-types@15.8.1/prop-types.js"></script>
    <script type="text/babel">
        class Person extends React.Component {
            static propTypes = {
                name: PropTypes.string.isRequired,
                gender: PropTypes.string,
                age: PropTypes.number,
                lmhSpeak: PropTypes.func
            }  

            static defaultProps = {
                gender: 'nt',
                age: 18
            }

            render() {
                const { name, gender, age } = this.props;
                return (
                    <ul>
                        <li>{name}</li>
                        <li>{gender}</li>
                        <li>{age + 1}</li>
                    </ul>
                )
            }
        }

        const p = { name: "clown", gender: "boy", age: 21 }
        const speak = () => console.log("speak");

        ReactDOM.render(<Person {...p} />, document.getElementById('test1'));
        ReactDOM.render(<Person name="lmh" age={21} lmhSpeak={speak} />, document.getElementById('test2'));
        ReactDOM.render(<Person name="rainy" age={18} gender="girl" />, document.getElementById('test3'));
    </script>
</body>
</html>
```
